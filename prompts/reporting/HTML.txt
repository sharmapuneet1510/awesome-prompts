function sendMessage() {
  const input = document.getElementById('userInput');
  const text = input.value.trim();
  if (!text) return;

  const messages = document.getElementById('messages');
  const bottomSuggestions = document.getElementById('bottom-suggestions');
  if (bottomSuggestions) bottomSuggestions.remove();

  // Add user message
  messages.insertAdjacentHTML('beforeend', createUserMessage(text));
  input.value = '';
  input.style.height = 'auto';

  // Show typing
  const typing = document.getElementById('typing');
  typing.style.display = 'flex';
  document.getElementById('sendBtn').disabled = true;

  messages.scrollTop = messages.scrollHeight;

  setTimeout(() => {
    typing.style.display = 'none';
    const reply = getResponse(text);
    messages.insertAdjacentHTML('beforeend', createAIMessage(reply));
    document.getElementById('sendBtn').disabled = false;
    messages.scrollTop = messages.scrollHeight;
  }, 1200 + Math.random() * 600);
}

function insertQuestion(text) {
  const input = document.getElementById('userInput');
  input.value = text;
  autoResize(input);
  input.focus();
}

function handleKey(e) {
  if (e.key === 'Enter' && !e.shiftKey) {
    e.preventDefault();
    sendMessage();
  }
}

function autoResize(el) {
  el.style.height = 'auto';
  el.style.height = Math.min(el.scrollHeight, 180) + 'px';
}
